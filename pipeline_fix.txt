#!/bin/bash
# Pipeline corrigido - verifica estrutura existente

cat > pipeline_fix.py << 'EOF'
import requests
import os
import psycopg2
import re

CSV_URL = 'https://dados.prefeitura.sp.gov.br/dataset/6588aef7-20ff-4cec-b1e1-6c06520240c0/resource/fd48b7dd-c5f1-4352-963f-f1e5ebc6d61b/download/contratos.csv'

print("=" * 80)
print("PIPELINE FIX - São Paulo 2024")
print("=" * 80)

conn = psycopg2.connect(os.getenv('DATABASE_URL'))
cur = conn.cursor()

# Verificar tabela existente
cur.execute("SELECT column_name FROM information_schema.columns WHERE table_name = 'sp_contratos'")
cols = cur.fetchall()
print(f"Colunas existentes: {[c[0] for c in cols]}")

# Dropar e recriar com schema correto
cur.execute("DROP TABLE IF EXISTS sp_contratos")
cur.execute("""
    CREATE TABLE sp_contratos (
        id SERIAL PRIMARY KEY,
        orgao TEXT,
        fornecedor TEXT,
        valor NUMERIC,
        importado_em TIMESTAMP DEFAULT NOW()
    )
""")
print("Tabela recriada")

print("\n[1/3] Baixando CSV...")
r = requests.get(CSV_URL, timeout=120)
lines = r.text.split('\n')

# Encontrar header
header_line = None
for i, line in enumerate(lines[:10]):
    if 'Fornecedor' in line and 'Valor' in line:
        header_line = i
        break

header = [h.strip().replace('"', '') for h in lines[header_line].split(';')]
fornecedor_idx = next((i for i, h in enumerate(header) if 'Fornecedor' in h), None)
valor_idx = next((i for i, h in enumerate(header) if 'Valor' in h), None)

print(f"Fornecedor: {fornecedor_idx}, Valor: {valor_idx}")

print("\n[2/3] Processando...")
dados = []
for line in lines[header_line + 1:]:
    if not line.strip():
        continue
    cols = [c.strip().replace('"', '') for c in line.split(';')]
    if len(cols) < 10:
        continue
    
    fornecedor = cols[fornecedor_idx] if len(cols) > fornecedor_idx else ''
    valor_str = cols[valor_idx] if len(cols) > valor_idx else ''
    
    # Limpar valor
    valor_str = valor_str.replace('"', '').replace('R$', '').replace('.', '').replace(',', '.').strip()
    try:
        valor = float(valor_str) if valor_str else None
    except:
        valor = None
    
    if fornecedor and valor and valor > 0:
        dados.append(('Prefeitura SP', fornecedor, valor))
    
    if len(dados) >= 25000:
        break

print(f"Processados: {len(dados)}")

print("\n[3/3] Inserindo...")
for d in dados:
    cur.execute("INSERT INTO sp_contratos (orgao, fornecedor, valor) VALUES (%s, %s, %s)", d)

conn.commit()

# Baseline
cur.execute("SELECT COUNT(*), COUNT(DISTINCT fornecedor), AVG(valor), MAX(valor) FROM sp_contratos")
row = cur.fetchone()

print("\n" + "=" * 80)
print("BASELINE SP 2024")
print("=" * 80)
print(f"Total: {row[0]:,}")
print(f"Fornecedores únicos: {row[1]:,}")
print(f"Valor médio: R$ {row[2]:,.2f}")
print(f"Valor máximo: R$ {row[3]:,.2f}")

# Top 5
cur.execute("SELECT fornecedor, COUNT(*), SUM(valor) FROM sp_contratos GROUP BY fornecedor ORDER BY COUNT(*) DESC LIMIT 5")
for i, r in enumerate(cur.fetchall(), 1):
    print(f"{i}. {r[0][:40]} - {r[1]} contratos")

print("=" * 80)
print("✓ CONCLUÍDO!")

conn.close()
EOF

python pipeline_fix.py
